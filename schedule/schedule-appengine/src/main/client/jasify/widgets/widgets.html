Booking via Jasify widget To embed it, use this code snippet: 

<!-- Container for widget -->
<div id="booking-via-jasify></div>
<!-- Creator script initializer (it registers window.createBookingViaJasify function) -->
<script src="booking-via-jasify.js"></script>
<!-- Library for resizing iframe to fit content -->
<script src="iframeResizer.min.js"></script>
<!-- Script that creates widget. It must be executed when #booking-via-jasify element is present on the page -->
<script>
window.createBookingViaJasify();
iFrameResize();
</script>
